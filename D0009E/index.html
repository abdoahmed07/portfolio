Same problem, three different data structures

All three implement a word dictionary with insert, lookup, and delete. The live demo is on the last card — the Python dict, which is the cleanest and most natural approach.

L3 · Approach 1
Word Book — Using Two Lists

Two synchronized lists: words[] and defs[]. The position of a word in one list always matches its definition in the other.

Parallel ListsIndex syncO(n) lookup
L3-1.py
def insert(words, defs):
    word = input("Enter word to insert: ").strip()
    definition = input("Enter definition: ").strip()
    if word in words:
        print(f"'{word}' already exists.")
        return
    words.append(word)
    defs.append(definition)
    print(f"Inserted '{word}': '{definition}'")

def lookup(words, defs):
    word = input("Enter word to lookup: ").strip()
    if word in words:
        i = words.index(word)
        print(f"You looked up {word}: {defs[i]}")
    else:
        print(f"'{word}' not found.")

def delete(words, defs):
    word = input("Enter word to delete: ").strip()
    if word in words:
        i = words.index(word)
        words.pop(i)
        defs.pop(i)
        print(f"Deleted '{word}'.")
    else:
        print(f"'{word}' not found.")

def main_lists():
    words, defs = [], []
    while True:
        choice = input("1.Insert 2.Lookup 3.Delete 4.Exit: ").strip()
        if   choice == "1": insert(words, defs)
        elif choice == "2": lookup(words, defs)
        elif choice == "3": delete(words, defs)
        elif choice == "4": break

def insert(entries):
    word = input("Enter word to insert: ").strip()
    definition = input("Enter definition: ").strip()
    for w, d in entries:
        if w == word:
            print(f"'{word}' already exists.")
            return
    entries.append((word, definition))
    print(f"Inserted '{word}': '{definition}'")

def lookup(entries):
    word = input("Enter word to lookup: ").strip()
    for w, d in entries:
        if w == word:
            print(f"You looked up {word}: {d}")
            return
    print(f"'{word}' not found.")

def delete(entries):
    word = input("Enter word to delete: ").strip()
    for i, (w, d) in enumerate(entries):
        if w == word:
            entries.pop(i)
            print(f"Deleted '{word}'.")
            return
    print(f"'{word}' not found.")

def main_tuples():
    entries = []

        if   choice == "1": insert(entries)
        elif choice == "2": lookup(entries)
        elif choice == "3": delete(entries)
        elif choice == "4": break

def insert(d):

    word = input("Enter word to lookup: ")

        print(f"'{word}' was not found.")

def delete(d):
    word = input("Enter word to delete: ")
    if word in d:
        del d[word]
        print(f"Deleted '{word}'.")

        print(f"'{word}' was not found.")

class phoneBook:
    def __init__(self):
        self.entries = {}  # number -> list of names (aliases)

    def add(self, name, number):
        if number not in self.entries:
            self.entries[number] = []
        if name in self.entries[number]:
            print(f"{name} already exists for {number}")
        else:
            self.entries[number].append(name)
            print(f"Added {name} with number {number}")

    def lookup(self, name):
        matches = [num for num, names in self.entries.items() if name in names]
        if not matches:         print(f"{name} not found")
        elif len(matches) == 1: print(f"{name}'s number is {matches[0]}")
        else:                   print(f"{name} has multiple numbers: {', '.join(matches)}")

    def alias(self, existing_name, new_name, number=None):
        matches = [num for num, names in self.entries.items() if existing_name in names]
        if not matches: print(f"{existing_name} not found"); return
        if number is None: number = matches[0]
        if new_name in self.entries[number]:
            print(f"{new_name} already exists for {number}")
        else:
            self.entries[number].append(new_name)
            print(f"Added alias {new_name} for {existing_name} ({number})")

    def change(self, name, new_number, old_number=None):
        matches = [num for num, names in self.entries.items() if name in names]
        if not matches: print(f"{name} not found"); return
        if old_number is None: old_number = matches[0]
        self.entries[new_number] = list(self.entries[old_number])
        del self.entries[old_number]
        print(f"Changed {name}'s number from {old_number} to {new_number}")

    def remove(self, name, number):
        if number not in self.entries or name not in self.entries[number]:
            print(f"{name} with {number} not found"); return
        self.entries[number].remove(name)
        if not self.entries[number]: del self.entries[number]
        print(f"Removed {name} ({number})")

    def list_entries(self):
        for num, names in self.entries.items():
            print(f"{num}: {', '.join(names)}")

    def run(self):
        commands = {
            "add":    lambda a: self.add(a[0], a[1])    if len(a)==2 else print("add <name> <number>"),
            "lookup": lambda a: self.lookup(a[0])        if len(a)==1 else print("lookup <name>"),
            "alias":  lambda a: self.alias(*a[:3])       if 2<=len(a)<=3 else print("alias <name> <alias>"),
            "change": lambda a: self.change(*a[:3])      if 2<=len(a)<=3 else print("change <name> <number>"),
            "remove": lambda a: self.remove(a[0], a[1]) if len(a)==2 else print("remove <name> <number>"),
            "list":   lambda a: self.list_entries(),
        }
        while True:
            parts = input("phoneBook> ").strip().split()
            if not parts: continue
            if parts[0] == "quit": break
            if parts[0] in commands: commands[parts[0]](parts[1:])
            else: print("Unknown command")

phoneBook().run()
