public class ArithmeticProblem {
    private final int operand1, operand2;
    private final char operator;

    public ArithmeticProblem(char op) {
        if (op != '+' && op != '-' && op != '*' && op != '/')
            throw new IllegalArgumentException("Operator is not legal.");
        operator = op;
        switch (operator) {
            case '+':
                operand1 = (int)(101 * Math.random());
                operand2 = (int)((101 - operand1) * Math.random());
                break;
            case '-':
                operand1 = 9 + (int)(92 * Math.random());
                operand2 = (int)((1 + operand1) * Math.random());
                break;
            case '*':
                operand1 = (int)(51 * Math.random());
                operand2 = (int)(13 * Math.random());
                break;
            default: // '/'
                int answer = (int)(13 * Math.random());
                operand2 = 1 + (int)(12 * Math.random());
                operand1 = operand2 * answer;
                break;
        }
    }

    public String getQuestion() {
        return String.format("What is %d %s %d ?", operand1, operator, operand2);
    }

    public int getAnswer() {
        switch (operator) {
            case '+': return operand1 + operand2;
            case '-': return operand1 - operand2;
            case '*': return operand1 * operand2;
            default:  return operand1 / operand2;
        }
    }
}

public class ArithmeticQuiz {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        System.out.println("Welcome to the Arithmetic Quiz!");
        int points = 0;

        for (int i = 0; i < 10; i++) {
            char[] ops = {'+', '-', '*', '/'};
            char op = ops[(int)(Math.random() * 4)];
            ArithmeticProblem problem = new ArithmeticProblem(op);

            System.out.println("Question " + (i + 1) + ":");
            System.out.println(problem.getQuestion());
            int userAnswer = in.nextInt();

            if (userAnswer == problem.getAnswer()) {
                System.out.println("Correct! You get 10 points.");
                points += 10;
            } else {
                System.out.println("Incorrect. Try again.");
                System.out.println(problem.getQuestion());
                userAnswer = in.nextInt();
                if (userAnswer == problem.getAnswer()) {
                    System.out.println("Correct! You get 5 points.");
                    points += 5;
                } else {
                    System.out.println("Incorrect. The correct answer is "
                        + problem.getAnswer() + ".");
                }
            }
        }
        System.out.println("Your total points: " + points + " out of 100.");

public class Spellcheck {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        WordList wordList = new WordList();
        System.out.println("Word list contains " + wordList.size() + " words.");

        while (true) {
            System.out.print("Enter a word (or 'exit'): ");
            String input = in.nextLine().trim().toLowerCase();
            if (input.equals("exit")) break;

            if (wordList.contains(input))
                System.out.println("'" + input + "' is spelled correctly.");
            else {
                System.out.println("'" + input + "' is NOT spelled correctly.");
                System.out.println("Did you mean:");
                delete_suggestion(input, wordList);
                insert_suggestion(input, wordList);
                swap_suggestion(input, wordList);
                swap_suggestion2(input, wordList);
                insert_space_suggestion(input, wordList);
            }
        }
    }

    // Delete one character at each position
    static void delete_suggestion(String input, WordList wl) {
        for (int i = 0; i < input.length(); i++) {
            String s = input.substring(0,i) + input.substring(i+1);
            if (wl.contains(s)) System.out.println("  - " + s);
        }
    }

    // Insert every letter at every position
    static void insert_suggestion(String input, WordList wl) {
        for (char c = 'a'; c <= 'z'; c++)
            for (int i = 0; i <= input.length(); i++) {
                String s = input.substring(0,i) + c + input.substring(i);
                if (wl.contains(s)) System.out.println("  - " + s);
            }
    }

    // Replace each character with every other letter
    static void swap_suggestion(String input, WordList wl) {
        for (int i = 0; i < input.length(); i++)
            for (char c = 'a'; c <= 'z'; c++) {
                if (c == input.charAt(i)) continue;
                String s = input.substring(0,i) + c + input.substring(i+1);
                if (wl.contains(s)) System.out.println("  - " + s);
            }
    }

    // Swap adjacent characters
    static void swap_suggestion2(String input, WordList wl) {
        for (int i = 0; i < input.length() - 1; i++) {
            char[] arr = input.toCharArray();
            char tmp = arr[i]; arr[i] = arr[i+1]; arr[i+1] = tmp;
            String s = new String(arr);
            if (wl.contains(s)) System.out.println("  - " + s);
        }
    }

    // Split into two valid words
    static void insert_space_suggestion(String input, WordList wl) {
        for (int i = 1; i < input.length(); i++) {
            String a = input.substring(0,i), b = input.substring(i);
            if (wl.contains(a) && wl.contains(b))
                System.out.println("  - " + a + " " + b);
        }
    }
}
WordList.java
public class WordList {
    private final String[] words;

    public WordList() {
        try {
            URL listLocation = WordList.class.getClassLoader()
                                             .getResource("unsorted_words.txt");
            Scanner in = new Scanner(listLocation.openStream());
            ArrayList<String> wordList = new ArrayList<>();
            while (in.hasNextLine())
                wordList.add(in.nextLine());
            words = wordList.toArray(new String[0]);
            in.close();
        } catch (Exception e) {
            throw new IllegalStateException("Can't load word list.");
        }
    }

    public boolean contains(String lowerCaseWord) {
        for (String s : words)
            if (s.equals(lowerCaseWord)) return true;
        return false;
    }

    public int size()            { return words.length; }
    public String get(int index) { return words[index]; }
}

MyArrayList.java
public class MyArrayList<E> implements List<E>, Cloneable, Serializable, RandomAccess {

    private E[] array;
    private int size;
    private static final int DEFAULT_CAPACITY = 10;

    // ── CONSTRUCTORS ──────────────────────────────────────────
    public MyArrayList(int initialCapacity) {
        if (initialCapacity < 0)
            throw new IllegalArgumentException("Illegal Capacity: " + initialCapacity);
        array = (E[]) new Object[initialCapacity];
        size = 0;
    }

    public MyArrayList() { this(DEFAULT_CAPACITY); }

    // ── §1: BASIC INFO ────────────────────────────────────────
    public int size()      { return size; }
    public boolean isEmpty() { return size == 0; }

    public void clear() {
        for (int i = 0; i < size; i++) array[i] = null;
        size = 0;
    }

    // ── §2: CAPACITY ──────────────────────────────────────────
    public void ensureCapacity(int minCapacity) {
        if (minCapacity > array.length) {
            int newCap = Math.max(array.length * 2 + 1, minCapacity);
            E[] newArr = (E[]) new Object[newCap];
            for (int i = 0; i < size; i++) newArr[i] = array[i];
            array = newArr;
        }
    }

    public void trimToSize() {
        if (size < array.length) {
            E[] newArr = (E[]) new Object[size];
            for (int i = 0; i < size; i++) newArr[i] = array[i];
            array = newArr;
        }
    }

    // ── §3: ADD ───────────────────────────────────────────────
    public void add(int index, E element) {
        if (index < 0 || index > size)
            throw new IndexOutOfBoundsException("Index: " + index + ", Size: " + size);
        ensureCapacity(size + 1);
        for (int i = size; i > index; i--) array[i] = array[i - 1];
        array[index] = element;
        size++;
    }

    public boolean add(E e) { add(size, e); return true; }

    // ── §4: READ / REPLACE ────────────────────────────────────
    public E get(int index) {
        if (index < 0 || index >= size)
            throw new IndexOutOfBoundsException("Index: " + index + ", Size: " + size);
        return array[index];
    }

    public E set(int index, E element) {
        if (index < 0 || index >= size)
            throw new IndexOutOfBoundsException("Index: " + index + ", Size: " + size);
        E old = array[index];
        array[index] = element;
        return old;
    }

    // ── §5: REMOVE ────────────────────────────────────────────
    public E remove(int index) {
        if (index < 0 || index >= size)
            throw new IndexOutOfBoundsException("Index: " + index + ", Size: " + size);
        E removed = array[index];
        removeRange(index, index + 1);
        return removed;
    }

    protected void removeRange(int fromIndex, int toIndex) {
        if (fromIndex < 0 || toIndex > size || fromIndex > toIndex)
            throw new IndexOutOfBoundsException();
        int count = toIndex - fromIndex;
        for (int i = toIndex; i < size; i++) array[i - count] = array[i];
        for (int i = size - count; i < size; i++) array[i] = null;
        size -= count;
    }

    // ── §6: SEARCH ────────────────────────────────────────────
    public int indexOf(Object o) {
        for (int i = 0; i < size; i++)
            if (o == null ? array[i] == null : o.equals(array[i])) return i;
        return -1;
    }

    public boolean remove(Object o) {
        int idx = indexOf(o);
        if (idx == -1) return false;
        remove(idx);
        return true;
    }

    public boolean contains(Object o) { return indexOf(o) != -1; }

    // ── §7: COPY ──────────────────────────────────────────────
    public Object clone() {
        MyArrayList<E> copy = new MyArrayList<>(size);
        for (int i = 0; i < size; i++) copy.array[i] = array[i];
        copy.size = this.size;
        return copy;
    }

    public Object[] toArray() {
        Object[] out = new Object[size];
        for (int i = 0; i < size; i++) out[i] = array[i];
        return out;
    }
}

public static void testAvAddindex() {
    data = new MyArrayList<Integer>(100);
    // negative index must throw
    try { data.add(-1, null); }

    assert data.get(1) == 10;

public static void testAvRemoveRange() {
    data = new MyArrayList<Integer>();
    insert(6); // [0, 1, 2, 3, 4, 5]
    data.removeRange(2, 4); // removes 2, 3
    // list is now: [0, 1, 4, 5]
    assert data.contains(0) && data.contains(1);
    assert !data.contains(2) && !data.contains(3);
    assert data.contains(4) && data.contains(5);
}

public static void testAvClone() {
    data = new MyArrayList<Integer>();
    insert(3); // [0, 1, 2]
    MyArrayList<Integer> clone = (MyArrayList<Integer>) data.clone();
    // clone has same values
    assert clone.get(0).equals(data.get(0));
    // changing clone does NOT affect original (own array)
    clone.set(1, 99);
    assert !data.get(1).equals(99);
}

Nivå.java
public class Nivå extends Observable {
    private ArrayList<Rum> rum;
    private Rum nuvarandeRum;

    public Nivå(Rum startrum, ArrayList<Rum> rum) {
        if (startrum == null || rum == null)
            throw new IllegalArgumentException("Start room and rooms cannot be null");
        this.rum = rum;
        this.nuvarandeRum = startrum;
        if (!this.rum.contains(startrum))
            throw new IllegalArgumentException("Start room is not included in the rooms");
        // Validate no overlap
        for (int i = 0; i < rum.size(); i++)
            for (int j = i + 1; j < rum.size(); j++)
                if (rum.get(i).getX() < rum.get(j).getX() + rum.get(j).getWall_width()
                    && rum.get(i).getX() + rum.get(i).getWall_width() > rum.get(j).getX()
                    && rum.get(i).getY() < rum.get(j).getY() + rum.get(j).getWall_height()
                    && rum.get(i).getY() + rum.get(i).getWall_height() > rum.get(j).getY())
                        throw new IllegalArgumentException("Two rooms overlap");
    }

    public ArrayList<Rum> getRum()  { return rum; }
    public Rum getNuvarandeRum()     { return nuvarandeRum; }

    public void hoppaÅt(Väderstreck v) {
        if (nuvarandeRum.finnsUtgångÅt(v)) {
            nuvarandeRum = nuvarandeRum.gångenÅt(v).getTill();
            setChanged();
            notifyObservers(); // triggers GUI.update() → repaint
        }
    }
}

public class Rum {
    private int x, y, höjd, bredd;
    private Color golvfärg;
    private Gång[] gångar = new Gång[GlobalaKonstanter.ANTAL_VÄDERSTRECK];

    public Rum(Color golvfärg, int bredd, int höjd, int övX, int övY) {
        this.golvfärg = golvfärg;
        this.höjd = höjd; this.bredd = bredd;
        this.x = övX;     this.y = övY;
    }

    public int getX()             { return x; }
    public int getY()             { return y; }
    public int getWall_height()   { return höjd; }
    public int getWall_width()    { return bredd; }
    public Color getFloor_color() { return golvfärg; }

    public boolean finnsUtgångÅt(Väderstreck v) {
        return gångar[v.index()] != null;
    }

    public Gång gångenÅt(Väderstreck v) {
        if (gångar[v.index()] == null)
            throw new IllegalArgumentException("No passage in that direction");
        return gångar[v.index()];
    }

    public static void kopplaIhop(Rum från, Väderstreck ut, Rum till, Väderstreck in) {
        if (från == null || till == null || ut == null || in == null)
            throw new IllegalArgumentException("Params cannot be null");
        if (från == till) throw new IllegalArgumentException("Same room");
        if (från.finnsUtgångÅt(ut)) throw new IllegalArgumentException("Already connected (från)");
        if (till.finnsUtgångÅt(in)) throw new IllegalArgumentException("Already connected (till)");
        från.gångar[ut.index()] = new Gång(från, ut, till, in);
        till.gångar[in.index()] = new Gång(till, in, från, ut);
    }
}

public class Målarduk extends JPanel {
    private final Nivå enNivå;

    public Målarduk(Nivå enNivå) {
        this.enNivå = enNivå;
        setBackground(MARKFÄRG);  // green background (Color.GREEN)
        setFocusable(true);
    }

    @Override
    protected void paintComponent(Graphics g) {
        super.paintComponent(g);
        for (Rum r : enNivå.getRum()) ritaRum(g, r);
        for (Rum r : enNivå.getRum()) ritaGångarFrånRum(g, r);
        ritaMarkörFörVarAnvändarenÄr(g);
    }

    private void ritaRum(Graphics g, Rum r) {
        Graphics2D g2 = (Graphics2D) g;
        g2.setColor(r.getFloor_color());
        g2.fillRect(r.getX(), r.getY(), r.getWall_width(), r.getWall_height());
        g2.setColor(VÄGGFÄRG); // black
        g2.setStroke(new BasicStroke(VÄGGTJOCKLEK)); // 8px walls
        g2.drawRect(r.getX(), r.getY(), r.getWall_width(), r.getWall_height());
    }

    private void ritaGångarFrånRum(Graphics g, Rum r) {
        for (Väderstreck v : Väderstreck.values())
            if (r.finnsUtgångÅt(v)) {
                Gång gång = r.gångenÅt(v);
                if (gång.getFrån() == r) ritaGång(g, gång); // draw each once
            }
    }

    private Punkt baspunkt(Rum r, Väderstreck v) {
        int midX = r.getX() + r.getWall_width() / 2;
        int midY = r.getY() + r.getWall_height() / 2;
        switch (v) {
            case NORR:   return new Punkt(midX, r.getY());
            case SÖDER:  return new Punkt(midX, r.getY() + r.getWall_height());
            case ÖSTER:  return new Punkt(r.getX() + r.getWall_width(), midY);
            case VÄSTER: return new Punkt(r.getX(), midY);
            default: throw new IllegalArgumentException();
        }
    }

    private Punkt pivotpunkt(Rum r, Väderstreck v) {
        Punkt b = baspunkt(r, v); int o = 7;
        switch (v) {
            case NORR:   return new Punkt(b.x(), b.y() - o);
            case SÖDER:  return new Punkt(b.x(), b.y() + o);
            case ÖSTER:  return new Punkt(b.x() + o, b.y());
            case VÄSTER: return new Punkt(b.x() - o, b.y());
            default: throw new IllegalArgumentException();
        }
    }

    private void ritaGång(Graphics g, Gång gång) {
        Graphics2D g2 = (Graphics2D) g;
        g2.setColor(GÅNGFÄRG); // brown
        g2.setStroke(new BasicStroke(VÄGGTJOCKLEK, CAP_ROUND, JOIN_ROUND));
        Punkt bf = baspunkt(gång.getFrån(), gång.getRiktningUtUrFrån());
        Punkt pf = pivotpunkt(gång.getFrån(), gång.getRiktningUtUrFrån());
        Punkt pt = pivotpunkt(gång.getTill(), gång.getRiktningInITill());
        Punkt bt = baspunkt(gång.getTill(), gång.getRiktningInITill());
        g2.drawLine(bf.x(), bf.y(), pf.x(), pf.y());
        g2.drawLine(pf.x(), pf.y(), pt.x(), pt.y());
        g2.drawLine(pt.x(), pt.y(), bt.x(), bt.y());
    }

    private void ritaMarkörFörVarAnvändarenÄr(Graphics g) {
        Rum r = enNivå.getNuvarandeRum();
        int cx = r.getX() + r.getWall_width() / 2;
        int cy = r.getY() + r.getWall_height() / 2;
        g.setColor(ANVÄNDARFÄRG); // black
        g.fillOval(cx - ANVÄNDARRADIE, cy - ANVÄNDARRADIE,
                   ANVÄNDARRADIE * 2, ANVÄNDARRADIE * 2);
    }
}

FIFO.java
public class FIFO implements Queue {
    private Object[] data;
    private int size, front, back;

    public FIFO(int capacity) {
        data  = new Object[capacity];
        size  = 0; front = 0; back = 0;
    }

    public int size()    { return size; }
    public boolean isEmpty() { return size == 0; }
    public int maxSize() { return data.length; }

    public Object first() {
        if (isEmpty()) throw new NoSuchElementException("Queue is empty");
        return data[front];
    }

    // Circular buffer: wrap index with modulo instead of shifting elements
    public void removeFirst() {
        if (isEmpty()) throw new NoSuchElementException("Queue is empty");
        data[front] = null;
        front = (front + 1) % data.length; // wrap around
        size--;
    }

    public void add(Object o) {
        if (size == data.length) throw new IllegalStateException("Queue is full");
        data[back] = o;
        back = (back + 1) % data.length;   // wrap around
        size++;
    }

    @Override
    public boolean equals(Object obj) {
        if (!(obj instanceof FIFO)) return false;
        FIFO other = (FIFO) obj;
        if (this.size != other.size) return false;
        for (int i = 0; i < size; i++) {
            Object a = this.data[(front + i) % data.length];
            Object b = other.data[(other.front + i) % other.data.length];
            if (a == null ? b != null : !a.equals(b)) return false;
        }
        return true;
    }
}

// Breadth-First Traversal using our FIFO queue
void bft(Graph graph, int startId) {
    FIFO queue = new FIFO(graph.nodeCount());
    boolean[] visited = new boolean[graph.nodeCount()];

    queue.add(startId);
    visited[startId] = true;

    while (!queue.isEmpty()) {
        int current = (int) queue.first();
        queue.removeFirst();               // dequeue front (FIFO order)

        System.out.println("Visiting node " + current);

        // enqueue all unvisited neighbours
        for (int neighbour : graph.neighbours(current)) {
            if (!visited[neighbour]) {
                visited[neighbour] = true;
                queue.add(neighbour);      // visit nearest nodes first
            }
        }
    }
}

// level by level — all distance-1 nodes before distance-2, etc.

public class BinOpButton extends CalculatorButton {
    private final char op;

    public int apply(int a, int b) {
        switch (op) {
            case '+': return a + b;
            case '-': return a - b;
            case '*': return a * b;
            case '/':
                if (b == 0) throw new ArithmeticException("Division by zero");
                return a / b; // integer division
            default: throw new IllegalStateException("Unknown operator: " + op);
        }
    }

    @Override
    protected void transition() {
        try {
            int current = Integer.parseInt(situation.display.getText());
            switch (situation.state) {
                case Input1:
                case HasResult:
                    // save left operand, set operator, wait for second number
                    situation.leftOperand = current;
                    situation.binaryOperator = this;
                    situation.state = State.OpReady;
                    break;
                case OpReady:
                    // just changing mind about operator
                    situation.binaryOperator = this;
                    break;
                case Input2:
                    // chain operation: compute first, then set new op
                    int result = situation.binaryOperator.apply(situation.leftOperand, current);
                    situation.display.setText("" + result);
                    situation.leftOperand = result;
                    situation.binaryOperator = this;
                    situation.state = State.OpReady;
                    break;
            }
        } catch (ArithmeticException ex) {
            situation.display.setText("0");
            situation.leftOperand = 0;
            situation.binaryOperator = null;
            situation.state = State.Input1;
        }
    }
}

Lab 6 — Design Notes

This document describes the initial design plan for the simulation. The UML diagram represents the intended architecture before implementation.

Package Structure

random

Generates random numbers. Provided by course instructors. ExponentialRandomStream for inter-arrival times. UniformRandomStream for service times.

simulation

Generic reusable simulation engine — not specific to car wash. Simulator drives the event loop. EventQueue uses a PriorityQueue ordered by time. Event is abstract with compareTo() for ordering. State holds time and stop flag.

carwash

Car wash–specific logic. CarWashState extends State. Each event type (Arrival, Departure, Close, Stop) extends Event and implements execute(). EventFactory centralises event creation to reduce coupling.

Key Design Decisions

  • EventFactory centralises event creation instead of letting events create each other directly — reduces coupling and keeps responsibilities clear.
  • Event implements Comparable<Event> so the PriorityQueue orders events by time automatically.
  • CarWashState extends State — the generic framework knows nothing about car washing; only the carwash package does.
  • SimulationView separates display from logic — state delegates printing to the view.
Event.java
// Abstract base — all events share a time and an execute() method
public abstract class Event implements Comparable<Event> {
    private final double time;

    protected Event(double time) { this.time = time; }
    protected double getTime()   { return time; }

    // Each subclass defines what happens when the event fires
    public abstract void execute(State state, EventQueue queue);

    // Natural ordering by time — used by PriorityQueue in EventQueue
    @Override
    public int compareTo(Event other) {
        return Double.compare(this.time, other.time);
    }
}
Simulator.java
public class Simulator {
    private final EventQueue eventQueue;
    private final State state;

    public void run() {
        // Core simulation loop — process events in time order
        while (!state.isStopped() && !eventQueue.isEmpty()) {
            Event e = eventQueue.next();     // always the earliest event
            state.setTime(e.getTime());      // advance simulation clock
            e.execute(state, eventQueue);    // fire the event
        }
    }
}

// EventQueue uses PriorityQueue<Event> — Event.compareTo() ensures
// the minimum-time event is always returned by next()
ArrivalEvent.java
class ArrivalEvent extends Event {
    private final EventFactory factory;

    @Override
    public void execute(State s, EventQueue q) {
        CarWashState state = (CarWashState) s;
        state.updateTimes(getTime()); // accumulate idle + queue times

        if (state.isClosed()) return; // car wash closed, ignore arrival
        state.incArrived();
        Car car = factory.newCar(getTime());

        // Priority: fast machine first, then slow, then queue, then reject
        if (state.getFreeFast() > 0) {
            state.takeFast();
            factory.scheduleDeparture(getTime(), car, true);
        } else if (state.getFreeSlow() > 0) {
            state.takeSlow();
            factory.scheduleDeparture(getTime(), car, false);
        } else if (state.getQueue().size() < state.getMaxQueue()) {
            state.getQueue().add(car);      // join the queue
        } else {
            state.incRejected();            // queue full → turned away
        }

        state.printRow(getTime(), "Arrive", String.valueOf(car.getId()));
        factory.scheduleNextArrival(getTime()); // schedule next car
    }
}
DepartureEvent.java
class DepartureEvent extends Event {
    private final Car car;
    private final boolean fast; // which machine type was used
    private final EventFactory factory;

    @Override
    public void execute(State s, EventQueue q) {
        CarWashState state = (CarWashState) s;
        state.updateTimes(getTime());
        state.incServed();
        state.printRow(getTime(), "Leave", String.valueOf(car.getId()));

        // Free the machine that was used
        if (fast) state.releaseFast();
        else      state.releaseSlow();

        // Immediately serve the next waiting car (if any)
        if (!state.getQueue().isEmpty()) {
            Car next = state.getQueue().poll();
            if (fast) state.takeFast();
            else      state.takeSlow();
            factory.scheduleDeparture(getTime(), next, fast);
        }
    }
}

package carwash;

import random.ExponentialRandomStream;
import random.UniformRandomStream;
import simulation.EventQueue;

/**
 * Factory class responsible for creating and scheduling all events in the
 * simulation.
 * Handles arrivals, departures, closing, and stopping the simulation.
 */
class EventFactory {
    // Random stream for car arrival times (exponential distribution)
    private final ExponentialRandomStream arrivalStream;
    // Random stream for fast machine wash times (uniform distribution)
    private final UniformRandomStream fastWashStream;
    // Random stream for slow machine wash times (uniform distribution)
    private final UniformRandomStream slowWashStream;

    // The event queue where all events are stored
    private EventQueue queue;

    // Counter for assigning unique IDs to cars
    private int nextCarId = 0;

    // The time when the car wash stops accepting new arrivals
    private final double closeTime;
    // The time when the entire simulation stops
    private final double stopTime;

    /**
     * Creates a new event factory with the specified random streams and time
     * settings.
     * 
     * @param arrivalStream  Random stream for car arrivals
     * @param fastWashStream Random stream for fast machine wash times
     * @param slowWashStream Random stream for slow machine wash times
     * @param closeTime      Time when the car wash stops accepting arrivals
     * @param stopTime       Time when the simulation stops
     */
    EventFactory(
            ExponentialRandomStream arrivalStream,
            UniformRandomStream fastWashStream,
            UniformRandomStream slowWashStream,
            double closeTime,
            double stopTime) {
        this.arrivalStream = arrivalStream;
        this.fastWashStream = fastWashStream;
        this.slowWashStream = slowWashStream;
        this.closeTime = closeTime;
        this.stopTime = stopTime;
    }

    /**
     * Initializes the factory with the event queue and state.
     * Schedules the initial events: first arrival, closing event, and stop event.
     * 
     * @param queue The event queue
     * @param state The car wash state
     */
    void init(EventQueue queue, CarWashState state) {
        this.queue = queue;

        // Schedule the first arrival
        scheduleNextArrival(0.0);
        // Schedule when the car wash closes to new arrivals
        queue.add(new CloseEvent(closeTime));
        // Schedule when the simulation stops
        queue.add(new StopEvent(stopTime));
    }

    /**
     * Schedules the next car arrival.
     * The time is determined by the arrival random stream.
     * 
     * @param currentTime The current simulation time
     */
    void scheduleNextArrival(double currentTime) {
        // Generate the time until the next arrival
        double dt = arrivalStream.next();
        double nextTime = currentTime + dt;

        // Only schedule if it happens before closing time
        if (nextTime < closeTime)
            queue.add(new ArrivalEvent(nextTime, this));
    }

    /**
     * Schedules a departure event for a car after it finishes washing.
     * 
     * @param currentTime The current simulation time
     * @param car         The car to wash
     * @param fast        True for fast machine, false for slow
     * @return The wash time that was generated
     */
    void scheduleDeparture(double currentTime, Car car, boolean fast) {
        // Generate a random wash time based on the machine type
        double washTime = fast ? fastWashStream.next() : slowWashStream.next();
        // Schedule the departure event
        queue.add(new DepartureEvent(currentTime + washTime, car, fast, this));
    }

    /**
     * Creates a new car with a unique ID.
     * 
     * @param arrivalTime The time this car arrived
     * @return A new Car object
     */
    Car newCar(double arrivalTime) {
        return new Car(nextCarId++, arrivalTime);
    }
}

EventQueue.java


package simulation;

import java.util.PriorityQueue;

/**
 * A queue that stores events in chronological order.
 * Events are always retrieved in order of their occurrence time.
 */
public class EventQueue {
    // Priority queue that orders events by time
    private final PriorityQueue<Event> pq = new PriorityQueue<>();

    /**
     * Adds an event to the queue.
     * The event will be stored in chronological order.
     * 
     * @param e The event to add
     */
    public void add(Event e) {
        pq.add(e);
    }

    /**
     * Gets and removes the next event from the queue.
     * 
     * @return The event that occurs next in time
     */
    protected Event next() {
        return pq.poll();
    }

    /**
     * Checks if the queue is empty.
     * 
     * @return true if no events are scheduled, false otherwise
     */
    protected boolean isEmpty() {
        return pq.isEmpty();
    }
}
